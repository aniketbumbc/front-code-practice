function longestConsecutive(nums) {
  // Remove duplicates and ignore empty items
  const set = new Set(nums.filter(n => n !== undefined));
  let longest = 0;

  for (const num of set) {
    // Only start counting when it's the beginning of a sequence
    if (!set.has(num - 1)) {
      let current = num;
      let count = 1;

      while (set.has(current + 1)) {
        current++;
        count++;
      }

      longest = Math.max(longest, count);
    }
  }

  return longest;
}

// Example usage
const arr = [4, 3, , 8, 1, 2, 6, 100, 9];
console.log(longestConsecutive(arr)); // Output: 4
